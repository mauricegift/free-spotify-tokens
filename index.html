const { token, token_type } = await
  fetch('https://stoken.giftedtech.co.ke/token.json').then(r => r.json());

// Search "Alan Walker Spectre"
const data = await fetch(
  'https://api.spotify.com/v1/search?q=Alan+Walker+Spectre&type=track&limit=5',
  { headers: { Authorization: `${token_type} ${token}` } }
).then(r => r.json());

console.log(data.tracks.items);
import requests

res = requests.get('https://stoken.giftedtech.co.ke/token.json').json()
token      = res['token']
token_type = res['token_type']  # 'Bearer'

# Search "Alan Walker Spectre"
data = requests.get(
    'https://api.spotify.com/v1/search',
    params={'q': 'Alan Walker Spectre', 'type': 'track', 'limit': 5},
    headers={'Authorization': f'{token_type} {token}'}
).json()

print(data['tracks']['items'])
# Fetch token then search "Alan Walker Spectre"
TOKEN=$(curl -s https://stoken.giftedtech.co.ke/token.json \
  | python3 -c "import sys,json; print(json.load(sys.stdin)['token'])")

curl -sG -H "Authorization: Bearer $TOKEN" \
  --data-urlencode "q=Alan Walker Spectre" \
  --data-urlencode "type=track" \
  --data-urlencode "limit=5" \
  "https://api.spotify.com/v1/search" | python3 -m json.tool
<?php
$res       = json_decode(file_get_contents('https://stoken.giftedtech.co.ke/token.json'), true);
$token      = $res['token'];
$token_type = $res['token_type']; // 'Bearer'

// Search "Alan Walker Spectre"
$url  = 'https://api.spotify.com/v1/search?' . http_build_query([
    'q' => 'Alan Walker Spectre', 'type' => 'track', 'limit' => 5
]);
$opts = ['http' => ['header' => "Authorization: $token_type $token\r\n"]];
$data = json_decode(file_get_contents($url, false, stream_context_create($opts)), true);
print_r($data['tracks']['items']);
?>
const axios = require('axios');

const { data: { token, token_type } } = await axios.get(
  'https://stoken.giftedtech.co.ke/token.json'
);

// Search "Alan Walker Spectre"
const { data } = await axios.get('https://api.spotify.com/v1/search', {
  params: { q: 'Alan Walker Spectre', type: 'track', limit: 5 },
  headers: { Authorization: `${token_type} ${token}` }
});

console.log(data.tracks.items);

Notes
Token expires ~1 hour after issue but is refreshed every 20 minutes, so it's always fresh when you fetch it.
This is an anonymous web-player token — works on all public endpoints. User-specific endpoints (liked songs, playlists) require OAuth login.
Runs 24/7 on a dedicated VPS by Gifted Tech. Check GitHub for uptime and commit history.
